        function submitForm(argument) {

            var form = ebaui.web.getForm( '.xxxx' );
            var isValid = form.validate();
            var onSucc = function( serverData ){
                alert( serverData );
            };

            if( !isValid ){

                alert( 'your form is not valid !!' );
                
            }else{

                form.submit( {

                    data : { t : 'testname' },
                    dataType : 'text',
                    success : onSucc

                } );

            }

        };
