;;(function( $,utils ){
    
    /**
     *  判断是否使用远程数据源
     *  @private
     *  @instance
     *  @memberof   ebaui.web.utils
     *  @method     _isUsingRemoteData
     *  @param      {Object|Array}          dataSource              - 如果dataSource是一个数组对象，则认为采用本地数据作为数据源；反之，如果dataSource包含了url属性，data属性（可选），怎认为是使用远程数据源
     *  @property   {String}                dataSource.url          - 服务端URL
     *  @property   {Object|Function}       dataSource.data         - 向服务器发起AJAX请求的时候，要发送给服务器的URL参数
     */
    utils['usingRemoteData'] = function( dataSource ){

        if( !dataSource ){
            throw 'dataSource can not be null!';
        }

        if( _.isArray( dataSource ) ){
            return false;
        }else if( dataSource && _.isObject( dataSource ) && 'url' in dataSource ){
            return true;
        }

        return false;

    };

})( jQuery,ebaui.web.utils );
