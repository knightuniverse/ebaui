Parameters:
Name Type Description

      //  初始化方式一
      var ns = ebaui.web;
      var btn = new ns.ListBox( $( '' ),{ title:'',id:'',name:'' } );

      $( '' ).listbox( { title:'',id:'',name:'' } )
      //  初始化方式三
      <input id="" title="" name="" data-role="listbox" data-options="{}" />
