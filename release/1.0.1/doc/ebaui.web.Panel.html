new Panel()

Author:

Example
      //  初始化方式一
      var ns = ebaui.web;
      var btn = new ns.Panel( $( '' ),{ title:'',id:'',name:'' } );
      //  初始化方式二
      $( '' ).panel( { title:'',id:'',name:'' } )
      //  初始化方式三
      <input id="" title="" name="" data-role="panel" data-options="{}" />
