Parameters:
Name Type Description

Example
      //  初始化方式一
      var ns = ebaui.web;
      var btn = new ns.Tabs( $( '' ),{ title:'',id:'',name:'' } );
      //  初始化方式二
      $( '' ).tabs( { title:'',id:'',name:'' } )
      //  初始化方式三
      <input id="" title="" name="" data-role="tabs" data-options="{}" />
