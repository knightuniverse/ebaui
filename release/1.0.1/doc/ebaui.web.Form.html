Parameters:
Name Type Description
iterator function 迭代器

     //  表单控件筛选条件
     var fitCondition = function( field ){
         //  some conditions to filter out form fields
     }; 
     form.eachField( function( field ){
         if( fitCondition( field ) ){
             //  设置表单控件的属性
             field.enabled( false );
             //  some other code
         }
     } );

Properties:
Name Type Description
settings.data Object 要提交到服务器的额外的数据
