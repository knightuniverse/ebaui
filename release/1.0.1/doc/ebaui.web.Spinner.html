Example
     //  初始化方式一
     var ns = ebaui.web;
     var btn = new ns.Spinner( $( '' ),{ title:'',id:'',name:'' } );
     //  初始化方式二
     $( '' ).spinner( { title:'',id:'',name:'' } )
